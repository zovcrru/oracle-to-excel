# Установка pytest (если ещё не установлен)
pip install pytest pytest-cov

# Запуск всех тестов
pytest test_env_config.py -v

# Запуск с покрытием кода
pytest test_env_config.py --cov=env_config --cov-report=html

# Запуск только тестов защиты паролей
pytest test_env_config.py -k "password" -v

# Запуск только критичных тестов логирования
python -m pytest tests/test_env_config.py::TestLoggingPasswordProtection -v

python -m pytest tests/test_env_config.py -v -s
Флаг -s покажет print-вывод и поможет понять, где найден файл.
